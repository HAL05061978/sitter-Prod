
DECLARE
    v_care_response RECORD;
    v_requester_id UUID;
    v_responder_id UUID;
    v_group_id UUID;
    v_requester_child_id UUID;
    v_responder_child_id UUID;
    v_reciprocal_scheduled_care_id UUID;
    v_original_scheduled_care_id UUID;
    v_responder_providing_id UUID;
    v_requester_needing_original_id UUID;
    v_requester_providing_reciprocal_id UUID;
    v_responder_needing_reciprocal_id UUID;
BEGIN
    RAISE NOTICE '=== ACCEPTING RECIPROCAL CARE RESPONSE ===';
    RAISE NOTICE 'Care Response ID: %', p_care_response_id;

    -- Get the care response details
    -- Accept 'submitted' status (after responder submits their offer)
    SELECT
        cr.*,
        crr.requester_id,
        crr.group_id,
        crr.child_id as requester_child_id,
        crr.requested_date,
        crr.start_time,
        crr.end_time
    INTO v_care_response
    FROM care_responses cr
    JOIN care_requests crr ON cr.request_id = crr.id
    WHERE cr.id = p_care_response_id
    AND cr.status = 'submitted';  -- Accept submitted responses (after submit_reciprocal_care_response)

    IF NOT FOUND THEN
        RAISE EXCEPTION 'Care response not found or not in submitted status. Response ID: %', p_care_response_id;
    END IF;

    RAISE NOTICE 'Found care response for request: %', v_care_response.request_id;
    RAISE NOTICE 'Requester: %, Responder: %', v_care_response.requester_id, v_care_response.responder_id;
    RAISE NOTICE 'Reciprocal date: % %-%',
        v_care_response.reciprocal_date,
        v_care_response.reciprocal_start_time,
        v_care_response.reciprocal_end_time;

    -- Update the care response status to accepted
    UPDATE care_responses
    SET
        status = 'accepted',
        updated_at = NOW()
    WHERE id = p_care_response_id;

    RAISE NOTICE 'Updated care response status to accepted';

    -- Create scheduled care for the responder (who will provide care for the original request)
    INSERT INTO scheduled_care (
        parent_id,
        group_id,
        care_date,
        start_time,
        end_time,
        care_type,
        status,
        notes,
        related_request_id,
        child_id
    ) VALUES (
        v_care_response.responder_id,
        v_care_response.group_id,
        v_care_response.requested_date,
        v_care_response.start_time,
        v_care_response.end_time,
        'provided',
        'confirmed',
        COALESCE(v_care_response.response_notes, 'Original request care provided by responder'),
        v_care_response.request_id,
        v_care_response.requester_child_id
    ) RETURNING id INTO v_original_scheduled_care_id;

    RAISE NOTICE 'Created scheduled care for responder: %', v_original_scheduled_care_id;

    -- Create scheduled care for the requester (who will need care during reciprocal time)
    INSERT INTO scheduled_care (
        parent_id,
        group_id,
        care_date,
        start_time,
        end_time,
        care_type,
        status,
        notes,
        related_request_id,
        child_id
    ) VALUES (
        v_care_response.requester_id,
        v_care_response.group_id,
        v_care_response.requested_date,
        v_care_response.start_time,
        v_care_response.end_time,
        'needed',
        'confirmed',
        COALESCE(v_care_response.response_notes, 'Original request care for requester'),
        v_care_response.request_id,
        v_care_response.requester_child_id
    ) RETURNING id INTO v_original_scheduled_care_id;

    RAISE NOTICE 'Created scheduled care for requester (original time): %', v_original_scheduled_care_id;

    -- Create scheduled care for the requester (who will provide reciprocal care)
    INSERT INTO scheduled_care (
        parent_id,
        group_id,
        care_date,
        start_time,
        end_time,
        care_type,
        status,
        notes,
        related_request_id,
        child_id
    ) VALUES (
        v_care_response.requester_id,
        v_care_response.group_id,
        v_care_response.reciprocal_date,
        v_care_response.reciprocal_start_time,
        v_care_response.reciprocal_end_time,
        'provided',
        'confirmed',
        'Reciprocal care provided by requester',
        v_care_response.request_id,
        v_care_response.reciprocal_child_id
    ) RETURNING id INTO v_reciprocal_scheduled_care_id;

    RAISE NOTICE 'Created reciprocal scheduled care for requester: %', v_reciprocal_scheduled_care_id;

    -- Create scheduled care for the responder (who will need care during reciprocal time)
    INSERT INTO scheduled_care (
        parent_id,
        group_id,
        care_date,
        start_time,
        end_time,
        care_type,
        status,
        notes,
        related_request_id,
        child_id
    ) VALUES (
        v_care_response.responder_id,
        v_care_response.group_id,
        v_care_response.reciprocal_date,
        v_care_response.reciprocal_start_time,
        v_care_response.reciprocal_end_time,
        'needed',
        'confirmed',
        'Reciprocal care for responder',
        v_care_response.request_id,
        v_care_response.reciprocal_child_id
    ) RETURNING id INTO v_reciprocal_scheduled_care_id;

    RAISE NOTICE 'Created scheduled care for responder (reciprocal time): %', v_reciprocal_scheduled_care_id;

    -- Add children to the scheduled care blocks
    -- In working environment, BOTH children are added to ALL 4 blocks

    -- Get the IDs of all 4 blocks we just created
    SELECT id INTO v_responder_providing_id
    FROM scheduled_care
    WHERE parent_id = v_care_response.responder_id
    AND care_date = v_care_response.requested_date
    AND start_time = v_care_response.start_time
    AND end_time = v_care_response.end_time
    AND care_type = 'provided'
    AND related_request_id = v_care_response.request_id
    ORDER BY created_at DESC LIMIT 1;

    SELECT id INTO v_requester_needing_original_id
    FROM scheduled_care
    WHERE parent_id = v_care_response.requester_id
    AND care_date = v_care_response.requested_date
    AND start_time = v_care_response.start_time
    AND end_time = v_care_response.end_time
    AND care_type = 'needed'
    AND related_request_id = v_care_response.request_id
    ORDER BY created_at DESC LIMIT 1;

    SELECT id INTO v_requester_providing_reciprocal_id
    FROM scheduled_care
    WHERE parent_id = v_care_response.requester_id
    AND care_date = v_care_response.reciprocal_date
    AND start_time = v_care_response.reciprocal_start_time
    AND end_time = v_care_response.reciprocal_end_time
    AND care_type = 'provided'
    AND related_request_id = v_care_response.request_id
    ORDER BY created_at DESC LIMIT 1;

    SELECT id INTO v_responder_needing_reciprocal_id
    FROM scheduled_care
    WHERE parent_id = v_care_response.responder_id
    AND care_date = v_care_response.reciprocal_date
    AND start_time = v_care_response.reciprocal_start_time
    AND end_time = v_care_response.reciprocal_end_time
    AND care_type = 'needed'
    AND related_request_id = v_care_response.request_id
    ORDER BY created_at DESC LIMIT 1;

    -- Add BOTH children to responder's providing block (original time)
    INSERT INTO scheduled_care_children (scheduled_care_id, child_id, providing_parent_id, notes, action_type)
    VALUES (v_responder_providing_id, v_care_response.requester_child_id, v_care_response.responder_id, 'Requester child needs care from responder', 'new');

    INSERT INTO scheduled_care_children (scheduled_care_id, child_id, providing_parent_id, notes, action_type)
    VALUES (v_responder_providing_id, v_care_response.reciprocal_child_id, v_care_response.responder_id, 'Responder child needs care from responder', 'new');

    -- Add BOTH children to requester's needing block (original time)
    INSERT INTO scheduled_care_children (scheduled_care_id, child_id, providing_parent_id, notes, action_type)
    VALUES (v_requester_needing_original_id, v_care_response.requester_child_id, v_care_response.responder_id, 'Requester child needs care from responder', 'new');

    INSERT INTO scheduled_care_children (scheduled_care_id, child_id, providing_parent_id, notes, action_type)
    VALUES (v_requester_needing_original_id, v_care_response.reciprocal_child_id, v_care_response.responder_id, 'Responder child needs care from responder', 'new');

    -- Add BOTH children to requester's providing block (reciprocal time)
    INSERT INTO scheduled_care_children (scheduled_care_id, child_id, providing_parent_id, notes, action_type)
    VALUES (v_requester_providing_reciprocal_id, v_care_response.requester_child_id, v_care_response.requester_id, 'Requester child needs care from requester', 'new');

    INSERT INTO scheduled_care_children (scheduled_care_id, child_id, providing_parent_id, notes, action_type)
    VALUES (v_requester_providing_reciprocal_id, v_care_response.reciprocal_child_id, v_care_response.requester_id, 'Responder child needs care from requester', 'new');

    -- Add BOTH children to responder's needing block (reciprocal time)
    INSERT INTO scheduled_care_children (scheduled_care_id, child_id, providing_parent_id, notes, action_type)
    VALUES (v_responder_needing_reciprocal_id, v_care_response.requester_child_id, v_care_response.requester_id, 'Requester child needs care from requester', 'new');

    INSERT INTO scheduled_care_children (scheduled_care_id, child_id, providing_parent_id, notes, action_type)
    VALUES (v_responder_needing_reciprocal_id, v_care_response.reciprocal_child_id, v_care_response.requester_id, 'Responder child needs care from requester', 'new');

    RAISE NOTICE 'Added both children to all 4 scheduled care blocks';

    -- IMPORTANT: Decline ALL other responses FIRST (both pending and submitted), before updating care_requests
    -- This is because there's a trigger sync_care_requests_on_response_change that runs on care_responses UPDATE
    -- If we update care_requests first, then decline responses, the trigger overwrites our status to 'declined'
    -- We decline both 'pending' (not yet responded) and 'submitted' (already responded) to prevent overbooking
    UPDATE care_responses
    SET
        status = 'declined',
        updated_at = NOW()
    WHERE request_id = v_care_response.request_id
    AND id != p_care_response_id
    AND status IN ('pending', 'submitted');

    RAISE NOTICE 'Rejected all other pending/submitted responses for this request';

    -- =====================================================
    -- NEW CODE: Send notifications
    -- =====================================================

    -- Send notification to the responder whose response was accepted
    -- This notification enables the calendar counter for the responder
    INSERT INTO notifications (
        id,
        user_id,
        type,
        title,
        message,
        data,
        is_read,
        created_at
    )
    SELECT
        gen_random_uuid(),
        v_care_response.responder_id,
        'care_accepted',
        'Reciprocal Care Response Accepted',
        'Your reciprocal care response has been accepted and the calendar has been updated.',
        jsonb_build_object(
            'care_response_id', p_care_response_id,
            'care_request_id', v_care_response.request_id,
            'requester_id', v_care_response.requester_id,
            'requested_date', v_care_response.requested_date,
            'start_time', v_care_response.start_time,
            'end_time', v_care_response.end_time,
            'reciprocal_date', v_care_response.reciprocal_date,
            'reciprocal_start_time', v_care_response.reciprocal_start_time,
            'reciprocal_end_time', v_care_response.reciprocal_end_time
        ),
        false,
        NOW();

    RAISE NOTICE 'Sent acceptance notification to responder';

    -- Send notification to responders whose responses were declined
    -- This notification enables the messages counter for non-accepted responders
    INSERT INTO notifications (
        id,
        user_id,
        type,
        title,
        message,
        data,
        is_read,
        created_at
    )
    SELECT
        gen_random_uuid(),
        cr.responder_id,
        'care_declined',
        'Your reciprocal response for ' || TO_CHAR(crq.requested_date, 'Mon DD, YYYY') || ' was not accepted',
        'The requester may have accepted a different response.',
        jsonb_build_object(
            'care_response_id', cr.id,
            'care_request_id', cr.request_id,
            'requested_date', crq.requested_date,
            'start_time', crq.start_time,
            'end_time', crq.end_time
        ),
        false,
        NOW()
    FROM care_responses cr
    JOIN care_requests crq ON cr.request_id = crq.id
    WHERE cr.request_id = v_care_response.request_id
    AND cr.id != p_care_response_id
    AND cr.status = 'declined'
    AND cr.updated_at >= NOW() - INTERVAL '1 minute';  -- Only send to recently declined responses

    RAISE NOTICE 'Sent decline notifications to other responders';

    -- =====================================================
    -- END NEW CODE
    -- =====================================================

    -- NOW update the care request status to completed (AFTER declining other responses)
    -- CRITICAL: Copy reciprocal date/time from care_responses to care_requests
    -- This must be LAST to avoid the trigger overwriting it
    UPDATE care_requests
    SET
        status = 'completed',
        responder_id = v_care_response.responder_id,
        reciprocal_parent_id = v_care_response.responder_id,
        reciprocal_child_id = v_care_response.reciprocal_child_id,
        reciprocal_date = v_care_response.reciprocal_date,
        reciprocal_start_time = v_care_response.reciprocal_start_time,
        reciprocal_end_time = v_care_response.reciprocal_end_time,
        reciprocal_status = 'accepted',
        updated_at = NOW()
    WHERE id = v_care_response.request_id;

    RAISE NOTICE 'Updated care request status to completed';

    RAISE NOTICE 'Reciprocal care response accepted successfully!';
    RETURN TRUE;

EXCEPTION
    WHEN OTHERS THEN
        RAISE NOTICE 'Error in accept_reciprocal_care_response: %', SQLERRM;
        RAISE EXCEPTION 'Failed to accept reciprocal care response: %', SQLERRM;
END;
